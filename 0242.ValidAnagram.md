### 想法二:

做法: 用array，初始化26個，用c-'a'所得的整數作為index讀取值

程式: c++
```c++
class Solution {
public:
    bool isAnagram(string s, string t) {
        int record[26] = {0};
        int index;
        for(int i = 0; i < s.size(); i++){
            index = s[i] - 'a';
            record[index]++;
        }
        for(int i = 0; i < t.size(); i++){
            index = t[i] - 'a';
            record[index]--;
        }
        for(int i = 0; i < 26; i++){
            if(record[i] != 0)
                return false;
        }
        return true;
    }
};
// success
```
